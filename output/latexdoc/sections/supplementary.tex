\subsection*{Supplementary Tables}

\begin{table}[htbp]
\centering
\caption{Sensitivity analysis: effect of random-effects structure on the condition coefficient for bruising depth. All specifications yield an identical point estimate ($\beta = -0.034$); standard errors and $p$-values vary substantially across specifications.}
\label{tab:supp-sensitivity}
\begin{tabular}{l S[table-format=-1.3] S[table-format=1.3] S[table-format=1.3]}
\toprule
{Random-effects specification} & {$\beta$ (condition)} & {SE} & {$p$-value} \\
\midrule
OLS (no random effects)                   & -0.034 & 0.084 & 0.692 \\
Random intercept (patient)                & -0.034 & 0.231 & 0.885 \\
Random intercept + random slope           & -0.034 & 0.231 & 0.885 \\
Two-level nested (quadrant within patient)& -0.034 & 0.231 & 0.876 \\
\bottomrule
\end{tabular}
\smallskip

\noindent\textit{Note:} OLS standard error (SE $= 0.084$) is 2.7$\times$ smaller than the LMM SE ($0.231$), confirming that ignoring clustering severely underestimates uncertainty. The condition estimate is invariant to the random-effects specification.
\end{table}

\begin{table}[htbp]
\centering

\label{tab:supp-grader}
\begin{tabular}{l l S[table-format=-1.3] S[table-format=1.3] S[table-format=1.3]}
\toprule
{Outcome} & {Grader treatment} & {$\beta$} & {SE} & {$p$-value} \\
\midrule

Depth  & Scores averaged by grader & -0.034 & 0.231 & 0.885 \\
Extent & Grader as fixed effect    &  0.029 & 0.146 & 0.844 \\
Extent & Scores averaged by grader &  0.029 & 0.146 & 0.844 \\
\bottomrule
\end{tabular}
\smallskip

\noindent\textit{Note:} The systematic Grader~1 bias (mean $+0.184$, $p < 0.001$) cancels out in the condition contrast because both graders assessed both conditions, explaining why the two approaches yield identical estimates.
\end{table}
